```python
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# データのロード
diamonds = sns.load_dataset("diamonds")

# 1. データ概要表示
print("データの先頭5行:")
print(diamonds.head(), "\n")

print("データの概要:")
print(diamonds.info(), "\n")

print("基本統計量:")
print(diamonds.describe(), "\n")

# 2. データ前処理
# 欠損値の確認
print("欠損値数:")
print(diamonds.isnull().sum(), "\n")

# 3. 可視化

# (1) カラットと価格の散布図
plt.figure(figsize=(8, 6))
sns.scatterplot(data=diamonds, x="carat", y="price", alpha=0.3)
plt.title("Carat vs Price")
plt.show()

# (2) カット別価格の箱ひげ図
plt.figure(figsize=(8, 6))
sns.boxplot(data=diamonds, x="cut", y="price")
plt.title("Price Distribution by Cut")
plt.show()

# (3) 色別価格のバイオリンプロット
plt.figure(figsize=(8, 6))
sns.violinplot(data=diamonds, x="color", y="price", inner="quartile")
plt.title("Price Distribution by Color")
plt.show()

# (4) 価格の分布ヒストグラム
plt.figure(figsize=(8, 6))
sns.histplot(data=diamonds, x="price", bins=50, kde=True)
plt.title("Price Distribution")
plt.show()

# (5) カット別・色別の平均価格ヒートマップ
avg_price = diamonds.pivot_table(index="cut", columns="color", values="price", aggfunc="mean")
plt.figure(figsize=(8, 6))
sns.heatmap(avg_price, annot=True, fmt=".0f", cmap="YlGnBu")
plt.title("Average Price by Cut and Color")
plt.show()
```
